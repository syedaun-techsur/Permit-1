```typescript
@Controller('dashboard')
@UseGuards(JwtAuthGuard, RolesGuard)
export class DashboardController {
  constructor(private readonly dashboardService: DashboardService) {}

  @Get('applicant')
  @Roles('applicant')
  getApplicantDashboard(@CurrentUser() user: User): Promise<ApplicantDashboardDto> {
    return this.dashboardService.getApplicantDashboard(user.id);
  }

  @Get('reviewer')
  @Roles('reviewer', 'admin')
  getReviewerDashboard(@CurrentUser() user: User): Promise<ReviewerDashboardDto> {
    return this.dashboardService.getReviewerDashboard(user.id);
  }

  @Get('admin')
  @Roles('admin')
  getAdminDashboard(): Promise<AdminDashboardDto> {
    return this.dashboardService.getAdminDashboard();
  }
}
```

**getApplicantDashboard(userId: string)**:

```sql
-- activeApplications
SELECT COUNT(*)::int FROM permit_applications
WHERE applicant_id = $1 AND status IN ('submitted','under_review','additional_info_needed');

-- actionRequired
SELECT COUNT(*)::int FROM permit_applications
WHERE applicant_id = $1 AND status = 'additional_info_needed';

-- unreadMessages (messages sent by others where applicant hasn't read)
SELECT COUNT(*)::int FROM messages m
JOIN permit_applications pa ON pa.id = m.application_id
WHERE pa.applicant_id = $1 AND m.sender_id != $1 AND m.is_read_by_applicant = FALSE;

-- recentApplications (last 5 by updated_at DESC)
SELECT
  id,
  reference_number,
  permit_type,
  status,
  updated_at,
  (
    SELECT COUNT(*)::int FROM messages msg
    WHERE msg.application_id = pa.id
      AND msg.sender_id != $1
      AND msg.is_read_by_applicant = FALSE
  ) AS unread_message_count
FROM permit_applications pa
WHERE applicant_id = $1
ORDER BY updated_at DESC
LIMIT 5;

-- pendingActions (additional_info_needed applications)
SELECT id, reference_number, decision_reason AS info_request_note, updated_at
FROM permit_applications
WHERE applicant_id = $1 AND status = 'additional_info_needed'
ORDER BY updated_at ASC;

-- activityFeed (last 10 notifications for this user)
SELECT id, type, title, body, application_id, is_read, created_at
FROM notifications
WHERE user_id = $1
ORDER BY created_at DESC
LIMIT 10;
```

Run each query via `this.dataSource.query(sql, [userId])`. Assemble and return `ApplicantDashboardDto`.

Note on `reference_number`: the permit_applications table may store it as `id` (UUID) or a formatted reference. Use `id` for the reference if `reference_number` column does not exist in the actual migration — check the Phase 2 migration file `backend/src/database/migrations/002_phase2_permits.sql` and use the actual column name.

**getReviewerDashboard(userId: string)**:

```sql
-- assignedApplications
SELECT COUNT(*)::int FROM permit_applications WHERE reviewer_id = $1;

-- awaitingResponse (applicant responded to info request — under_review after being additional_info_needed)
-- Use status = 'additional_info_needed' AND last history entry shows applicant responded
-- Simpler: count additional_info_needed where there's a history row to_status='under_review' after the info request
-- Use this query matching planning spec:
SELECT COUNT(*)::int FROM permit_applications
WHERE reviewer_id = $1
  AND status = 'additional_info_needed'
  AND id IN (
    SELECT application_id FROM permit_status_history
    WHERE to_status = 'under_review'
    ORDER BY created_at DESC
    LIMIT 1
  );
-- If info_response_at column does not exist, fall back to: count where status='additional_info_needed' AND reviewer_id=$1

-- unassignedInPool
SELECT COUNT(*)::int FROM permit_applications
WHERE reviewer_id IS NULL AND status = 'submitted';

-- unreadMessages across reviewer's assigned applications
SELECT COUNT(*)::int FROM messages m
JOIN permit_applications pa ON pa.id = m.application_id
WHERE pa.reviewer_id = $1 AND m.sender_id != $1 AND m.is_read_by_reviewer = FALSE;

-- priorityQueue (top 10 by priority)
SELECT
  id,
  reference_number,
  permit_type,
  status,
  submission_date,
  EXTRACT(DAY FROM NOW() - submission_date)::int AS days_since_submission,
  (
    SELECT u.full_name FROM users u
    WHERE u.id = pa.applicant_id
  ) AS applicant_name,
  (
    SELECT COUNT(*)::int FROM messages msg
    WHERE msg.application_id = pa.id
      AND msg.sender_id != $1
      AND msg.is_read_by_reviewer = FALSE
  ) AS unread_message_count
FROM permit_applications pa
WHERE (reviewer_id = $1 OR (reviewer_id IS NULL AND status = 'submitted'))
  AND status NOT IN ('approved','rejected','draft')
ORDER BY
  CASE status
    WHEN 'additional_info_needed' THEN 1
    WHEN 'submitted' THEN 2
    WHEN 'under_review' THEN 3
    ELSE 4
  END,
  submission_date ASC NULLS LAST
LIMIT 10;

-- atRiskApplications (under_review, assigned to reviewer, no status change for > 3 days)
SELECT
  id, reference_number, permit_type, status, submission_date,
  EXTRACT(DAY FROM NOW() - submission_date)::int AS days_since_submission,
  (SELECT u.full_name FROM users u WHERE u.id = pa.applicant_id) AS applicant_name,
  0 AS unread_message_count
FROM permit_applications pa
WHERE reviewer_id = $1
  AND status = 'under_review'
  AND updated_at < NOW() - INTERVAL '3 days'
ORDER BY updated_at ASC
LIMIT 5;

-- activityFeed (last 10 audit_logs for reviewer's assigned apps)
SELECT al.id, al.action, al.metadata, al.created_at,
       u.full_name AS actor_name, u.role AS actor_role,
       al.application_id
FROM audit_logs al
JOIN users u ON u.id = al.actor_id
WHERE al.application_id IN (
  SELECT id FROM permit_applications WHERE reviewer_id = $1
)
ORDER BY al.created_at DESC
LIMIT 10;
```

**getAdminDashboard()**:

```sql
-- totalApplications
SELECT COUNT(*)::int FROM permit_applications;

-- activeApplications (not in terminal state)
SELECT COUNT(*)::int FROM permit_applications
WHERE status NOT IN ('approved','rejected','draft');

-- submittedThisWeek
SELECT COUNT(*)::int FROM permit_applications
WHERE submission_date >= NOW() - INTERVAL '7 days';

-- decisionsThisWeek (approved or rejected this week)
SELECT COUNT(*)::int FROM permit_applications
WHERE status IN ('approved','rejected')
  AND updated_at >= NOW() - INTERVAL '7 days';

-- statusDistribution
SELECT status::text, COUNT(*)::int AS count
FROM permit_applications
GROUP BY status
ORDER BY count DESC;

-- reviewerWorkload
SELECT
  u.id AS reviewer_id,
  u.full_name AS reviewer_name,
  COUNT(pa.id) FILTER (WHERE pa.status NOT IN ('approved','rejected','draft'))::int AS assigned,
  COUNT(pa.id) FILTER (WHERE pa.status = 'under_review')::int AS under_review,
  COUNT(pa.id) FILTER (WHERE pa.status = 'additional_info_needed')::int AS additional_info_needed,
  COUNT(pa.id) FILTER (WHERE pa.status IN ('approved','rejected') AND pa.updated_at >= NOW() - INTERVAL '7 days')::int AS decided_this_week
FROM users u
LEFT JOIN permit_applications pa ON pa.reviewer_id = u.id
WHERE u.role = 'reviewer' AND u.is_active = TRUE
GROUP BY u.id, u.full_name
ORDER BY assigned DESC;

-- recentActivity (last 20 audit log entries system-wide)
SELECT al.id, al.action, al.metadata, al.created_at, al.application_id,
       u.full_name AS actor_name, u.role AS actor_role
FROM audit_logs al
JOIN users u ON u.id = al.actor_id
ORDER BY al.created_at DESC
LIMIT 20;
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→API | JWT-authenticated HTTP requests from browser to GET /dashboard/* endpoints; user id extracted from validated token, not from request body |
| db→render | Aggregated DB data returned in JSON response — stored text values (permit_type, full_name, action) reflected back to caller |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|

```typescript
  list: (applicationId: string, cursor?: string) =>
    axiosInstance.get<MessageListResponse>(`/permits/${applicationId}/messages`, { params: cursor ? { cursor } : {} }),
```

**`frontend/src/hooks/useMessages.ts`** — 30-second polling hook:

```typescript
export function useMessages(applicationId: string) {
  const [messages, setMessages] = useState<Message[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchMessages = useCallback(async () => {
    try {
      const res = await messagesApi.list(applicationId);
      setMessages(res.data.data);
    } catch (e) {
      setError('Failed to load messages');
    } finally {
      setLoading(false);
    }
  }, [applicationId]);

  useEffect(() => {
    fetchMessages();
    const interval = setInterval(fetchMessages, 30_000);
    return () => clearInterval(interval);
  }, [fetchMessages]);

  const sendMessage = useCallback(async (body: string) => {
    const res = await messagesApi.send(applicationId, body);
    setMessages((prev) => [...prev, res.data]);
    return res.data;
  }, [applicationId]);

  const markAllRead = useCallback(async (messageIds: string[]) => {
    await Promise.all(messageIds.map((id) => messagesApi.markRead(applicationId, id)));
  }, [applicationId]);

  return { messages, loading, error, sendMessage, markAllRead, refetch: fetchMessages };
}
```

```typescript
    setNotifications(res.data.data);

    setNotifLoading(false);

const markOneRead = useCallback(async (notifId: string) => {
  await notificationsApi.markOneRead(notifId);
  setNotifications((prev) => prev.map((n) => n.id === notifId ? { ...n, isRead: true } : n));
  // Refresh unread count
  fetchUnreadCount();
}, [fetchUnreadCount]);

const markAllRead = useCallback(async () => {
  await notificationsApi.markAllRead();
  setNotifications((prev) => prev.map((n) => ({ ...n, isRead: true })));
  setUnreadCount(0);
}, []);

return { unreadCount, notifications, notifLoading, loadNotifications, markOneRead, markAllRead };
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| db→render | Message body stored in DB and rendered in MessageBubble — XSS risk if rendered via dangerouslySetInnerHTML |
| client→API | Axios requests with Bearer token sent to messaging and notifications endpoints |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-15 | Tampering (XSS) | MessageBubble rendering `message.body` from DB | mitigate | MessageBubble renders `message.body` via React's `{message.body}` JSX interpolation (never `dangerouslySetInnerHTML`); React escapes all string content by default, preventing stored XSS |
| T-03-16 | Tampering (XSS) | NotificationPanel rendering `notification.body` | mitigate | Notification body rendered via `{notification.body}` JSX interpolation; same React escaping protection |
| T-03-17 | Information disclosure | Token exposure in Axios requests | transfer | Axios instance from Phase 2 attaches token from Zustand auth store (in-memory, not localStorage for access token); refresh token management is in httpOnly cookie — standard JWT pattern; risk owned by Phase 1 auth design |
| T-03-18 | Elevation of privilege | Reviewer-only attach button visible to applicants | mitigate | `MessageComposer` conditionally renders the paperclip button only when `isReviewer = true` prop is passed; the backend enforces RBAC independently — UI is defense-in-depth only |
</threat_model>

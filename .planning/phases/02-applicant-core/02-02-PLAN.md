**Step 3 — Create `backend/src/documents/s3.service.ts`** using the `minio` npm package (install: `npm install minio`):

```typescript
import { Injectable, Logger } from '@nestjs/common';
import * as Minio from 'minio';

@Injectable()
export class S3Service {
  private readonly client: Minio.Client;
  private readonly bucket: string;
  private readonly logger = new Logger(S3Service.name);

  constructor() {
    this.bucket = process.env.MINIO_BUCKET_NAME || 'permits-documents';
    this.client = new Minio.Client({
      endPoint: process.env.MINIO_ENDPOINT || 'localhost',
      port: parseInt(process.env.MINIO_PORT || '9000', 10),
      useSSL: process.env.MINIO_USE_SSL === 'true',
      accessKey: process.env.MINIO_ACCESS_KEY || 'minioadmin',
      secretKey: process.env.MINIO_SECRET_KEY || 'minioadmin',
    });
  }

  /**
   * Returns a presigned PUT URL valid for expiresInSeconds (default 900 = 15 min).
   * storageKey format: '{applicationId}/{uuid}-{filename}'
   */
  async getPresignedPutUrl(
    storageKey: string,
    expiresInSeconds = 900,
  ): Promise<string> {
    return this.client.presignedPutObject(this.bucket, storageKey, expiresInSeconds);
  }

  /**
   * Returns a presigned GET URL valid for expiresInSeconds (default 900 = 15 min).
   */
  async getPresignedGetUrl(
    storageKey: string,
    expiresInSeconds = 900,
  ): Promise<string> {
    return this.client.presignedGetObject(this.bucket, storageKey, expiresInSeconds);
  }

  /**
   * Schedules storage object deletion (fire-and-forget, async).
   * Called by soft-delete flow — do not await in controller.
   */
  async scheduleDelete(storageKey: string): Promise<void> {
    try {
      await this.client.removeObject(this.bucket, storageKey);
    } catch (err) {
      this.logger.warn(`Failed to delete object ${storageKey}: ${err}`);
    }
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→API | filename, mimeType, sizeBytes, storageKey from untrusted request body entering document handlers |
| client→MinIO | Presigned PUT URL consumed directly by browser; URL is signed with HMAC — MinIO validates |
| db→render | Document storage_key returned in presigned GET URL — must not leak other applicants' keys |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-08 | Elevation of privilege | GET /permits/:id/documents/:docId/url — IDOR to another applicant's document | mitigate | `DocumentsService.getDocumentUrl()` verifies `application.applicant_id === req.user.id` before generating presigned URL; application lookup includes ownership check before doc lookup |
| T-02-09 | Tampering | POST /permits/:id/documents — client supplies arbitrary storageKey to register a key they did not actually upload | mitigate | storageKey in `RegisterDocumentDto` is trusted only after the frontend has performed the presigned PUT (server does not re-verify upload completion in Phase 2, accepted residual risk — full verification via MinIO event hook is Phase 3 hardening); application and ownership verified before registration |
| T-02-10 | Denial of service | Mass document uploads to exceed 100 MB or 20-file limits | mitigate | `DocumentsService.getUploadUrl()` counts existing docs (≥ 20 → 422) and sums size_bytes (> 100 MB → 422) before returning presigned URL; checked server-side, not client-side only |
| T-02-11 | Path traversal | client-controlled filename used as storageKey component | mitigate | storageKey generated server-side as `${applicationId}/${randomUUID()}-${sanitized_filename}`; client-supplied filename is only appended after a UUID prefix that prevents directory traversal; `dto.filename` max 255 chars validated by class-validator |
| T-02-12 | Information disclosure | MinIO credentials in logs or response bodies | mitigate | S3Service reads credentials from env vars (MINIO_ACCESS_KEY, MINIO_SECRET_KEY) only; presigned URLs embed HMAC signatures, not raw credentials; NestJS response serialization does not include S3Service internals |
| T-02-13 | Tampering | DELETE /permits/:id/documents/:docId on a submitted application | mitigate | `DocumentsService.softDelete()` checks `app.status in ['draft','additional_info_needed']`; throws ForbiddenException otherwise; soft-delete never physically removes from MinIO immediately (scheduleDelete is fire-and-forget, providing audit trail window) |
</threat_model>

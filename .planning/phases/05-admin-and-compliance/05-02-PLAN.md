**Step 2 — `frontend/src/api/admin.api.ts`:**

Use the existing Axios instance (from Phase 1 — `src/api/axios.ts` or `src/lib/axios.ts`). All endpoints are admin-only; the JWT interceptor handles auth headers.

```typescript
import axiosInstance from './axios'; // adjust to actual Phase 1 path
import type {
  AdminPermit, AdminUser, AuditLogEntry, PaginatedResponse, AuditLogResponse,
  AdminPermitsQuery, AdminUsersQuery, AuditLogQuery,
  CreateUserPayload, UpdateUserPayload, AssignReviewerPayload, ReviewerOption,
} from '../types/admin.types';

export const adminApi = {
  // PERM-07
  getAllPermits: (params: AdminPermitsQuery) =>
    axiosInstance.get<PaginatedResponse<AdminPermit>>('/admin/permits', { params }).then(r => r.data),

  // ADMN-01
  getUsers: (params: AdminUsersQuery) =>
    axiosInstance.get<PaginatedResponse<AdminUser>>('/admin/users', { params }).then(r => r.data),
  getUserById: (userId: string) =>
    axiosInstance.get<AdminUser>(`/admin/users/${userId}`).then(r => r.data),
  createUser: (payload: CreateUserPayload) =>
    axiosInstance.post<AdminUser>('/admin/users', payload).then(r => r.data),
  updateUser: (userId: string, payload: UpdateUserPayload) =>
    axiosInstance.patch<AdminUser>(`/admin/users/${userId}`, payload).then(r => r.data),

  // ADMN-02 — assign reviewer (existing Phase 3 endpoint)
  assignReviewer: (permitId: string, payload: AssignReviewerPayload) =>
    axiosInstance.patch<AdminPermit>(`/permits/${permitId}/assign-reviewer`, payload).then(r => r.data),

  // Reviewer list for dropdown (reuse GET /admin/users?role=reviewer&isActive=true)
  getActiveReviewers: () =>
    axiosInstance.get<PaginatedResponse<AdminUser>>('/admin/users', { params: { role: 'reviewer', isActive: true, limit: 100 } }).then(r => r.data),

  // ADMN-03
  getAuditLog: (params: AuditLogQuery) =>
    axiosInstance.get<AuditLogResponse>('/admin/audit-log', { params }).then(r => r.data),

  // CSV export — returns the URL so the page can trigger browser download
  getExportUrl: (params: AuditLogQuery): string => {
    const base = axiosInstance.defaults.baseURL ?? '';
    const qs = new URLSearchParams(Object.entries(params).filter(([, v]) => v !== undefined) as [string, string][]).toString();
    return `${base}/admin/audit-log/export${qs ? '?' + qs : ''}`;
  },
};
```

| db→render | Audit log details (JSON field), user emails, permit data rendered in tables — XSS risk if not escaped |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-08 | Information disclosure | `adminApi.getExportUrl` — JWT in URL query param for CSV download | mitigate | Use Axios blob download (`responseType: 'blob'`) instead of bare anchor href; token stays in Authorization header, never in URL, in `AuditLogPage.tsx::handleExport` |
| T-05-09 | Elevation of privilege | Sidebar admin section + route guard | mitigate | Admin sidebar items rendered only when `user.role === 'admin'` in `Sidebar.tsx`; router-level `ProtectedRoute role="admin"` guard in `router/index.tsx` blocks direct URL navigation; backend enforces 403 independently (defense in depth) |
| T-05-10 | Information disclosure (XSS via audit log details) | `AuditLogPage.tsx` — `details` JSON rendered in table cell | mitigate | Render `details` field as `JSON.stringify(entry.details, null, 2)` inside a `<pre>` with `white-space: pre-wrap`; NEVER use `dangerouslySetInnerHTML`; React's JSX renderer escapes all string values |
| T-05-11 | Information disclosure | `AssignReviewerModal` — reviewer list with workload counts | accept | Reviewer names and workload counts are not sensitive; admin role already has access to this information; residual risk owned by product |
| T-05-12 | Tampering (open redirect) | `AuditLogPage` applicationRef link to application detail | mitigate | Application detail links constructed as `/permits/${entry.applicationId}` using the trusted `applicationId` UUID from backend response — never from raw `applicationRef` string; React Router `<Link to={...}>` does not perform external navigation |

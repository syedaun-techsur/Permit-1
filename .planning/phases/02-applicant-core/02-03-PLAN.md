export const permitsApi = {
  createPermit: (data: CreatePermitPayload) =>
    apiClient.post<PermitApplication>('/permits', data),

    apiClient.patch<PermitApplication>(`/permits/${id}`, data),

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-14 | Information disclosure | PermitListPage renders permits fetched from GET /permits | transfer | Server-side enforcement in PermitsService.listApplications() (02-01) filters by applicant_id; frontend consumes only what the authenticated API returns |
| T-02-15 | Tampering | Auto-save PATCH /permits/:id sends partial form payload | transfer | Backend UpdatePermitDto (02-01) uses PartialType(CreatePermitDto) — no status field accepted; ownership enforced server-side |
| T-02-16 | Denial of service | Auto-save fires on every keystroke if debounce not implemented | mitigate | `PermitFormPage.tsx` implements 2000ms debounce before PATCH call; at most 1 PATCH per 2 seconds per user session |
| T-02-17 | Repudiation | Auto-save failures silently dropped | mitigate | `PermitFormPage.tsx` implements 3× exponential retry (1s, 2s, 4s); after 3 failures displays warning toast via Phase 1 Toast primitive — user is informed |
| T-02-18 | Elevation of privilege | Frontend route /permits/:id/edit accessible without authentication | transfer | Phase 1 ProtectedRoute wrapper redirects unauthenticated users to /login; all permit routes nested under ProtectedRoute in router/index.tsx |
</threat_model>
